import java.util.*;

public class Solution {
    public int[] solution(int []arr) {
        int size = arr.length;
        ArrayList<Integer> result = new ArrayList<Integer>();
        result.add(arr[0]);
        for(int i = 1; i < size; i++){
            if(arr[i - 1] != arr[i]){
                result.add(arr[i]);
            }
        }
        int result_size = result.size(), count = 0;
        int[] answer = new int[result_size];
        for(Integer i : result){
            answer[count++] = i;
        }
        return answer;
    }
}
