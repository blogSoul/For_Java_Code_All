class Solution {
    public int[] solution(int brown, int yellow) {
        int[] answer = {0, 0};
        int range = (int)Math.sqrt(yellow);
        for(int i = 1; i <= range; i++){
            if(yellow % i == 0 && 2 * ((yellow / i) + i) + 4 == brown){
                answer[0] = yellow / i + 2;
                answer[1] = i + 2;
                break;
            }
        }
        return answer;
    }
}
