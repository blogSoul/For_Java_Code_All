import java.util.*;

class Solution {
    public int answer;
    public int[] arr;
    public boolean[] visited;
    public HashMap<Character, Integer> map;
    public void dfs(int count, String[] data){
        if(count == 8){ // data ì™„ë£Œ
            int data_size = data.length;
            for(int i = 0; i < data_size; i++){
                String order = data[i];
                int a = arr[map.get(order.charAt(0))];
                int b = arr[map.get(order.charAt(2))];
                char compare = order.charAt(3);
                int size = order.charAt(4) - '0';
                if(compare == '<'){
                    if(Math.abs(a - b) - 1 >= size){
                        return;
                    } else {
                        continue;
                    }
                } else if(compare == '>'){
                    if(Math.abs(a - b) - 1 <= size){
                        return;
                    } else {
                        continue;
                    }
                } else if(compare == '='){
                    if(Math.abs(a - b) - 1 != size){
                        return;
                    } else {
                        continue;
                    }
                }
            }
            answer++;
            return;
        }

        for(int i = 0; i < 8; i++){
            if(!visited[i]){
                visited[i] = true;
                arr[count] = i;
                dfs(count + 1, data);
                visited[i] = false;
            }
        }
    }
    public int solution(int n, String[] data) {
        int data_size = data.length;
        arr = new int[8];
        visited = new boolean[8];
        map = new HashMap<Character, Integer>();
        map.put('A', 0);
        map.put('C', 1);
        map.put('F', 2);
        map.put('J', 3);
        map.put('M', 4);
        map.put('N', 5);
        map.put('R', 6);
        map.put('T', 7);
        answer = 0;
        dfs(0, data);
        return answer;
    }
}
