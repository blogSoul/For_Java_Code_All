class Solution {
    public int x, y, number, size;
    public int[][] graph;
    public void initCycle(int n) {
        while(x < n){
            if(graph[x][y] != 0) {
                break;
            }
            graph[x++][y] = number++;
        }
        while(y < n){
            if(graph[x][y] != 0) {
                break;
            }
            graph[x][y++] = number++;
        }
        while(x >= 0 && y >= 0){
            if(graph[x][y] != 0) {
                break;
            }
            graph[x--][y--] = number++;
        }
        x += 1;
        y += 1;
    }
    public void Cycle() {
        while(true){
            if(graph[x + 1][y] != 0) {
                break;
            } else {
                x++;
            }
            graph[x][y] = number++;
        }
        while(true){
            if(graph[x][y + 1] != 0) {
                break;
            } else {
                y++;
            }
            graph[x][y] = number++;
        }
        while(true){
            if(graph[x - 1][y - 1] != 0) {
                break;
            } else {
                x--;
                y--;
            }
            graph[x][y] = number++;
        }
    }
    public int[] solution(int n) {
        x = 0; y = 0;
        size = n * (n + 1) / 2; number = 1;
        graph = new int[n][n];
        int cnt = 0;
        initCycle(n - 1);
        while(cnt++ < n - 3){
            Cycle();
        }
        int[] answer = new int[size];
        cnt = 0;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(graph[i][j] != 0){
                    answer[cnt++] = graph[i][j];
                }
            }
        }
        return answer;
    }
}
